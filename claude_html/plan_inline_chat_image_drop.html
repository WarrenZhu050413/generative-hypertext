    const handleSendMessage = async () => {
  if ((!inputValue.trim() && pendingImages.length === 0) || isStreaming) return;

  const userMessage: Message = {
    role: 'user',
    content: inputValue.trim() || '(Image attached)',
    images: pendingImages.length > 0 ? [...pendingImages] : undefined
  };

  setInputValue('');
  setPendingImages([]); // Clear pending images

  const newMessages = [...messages, userMessage];
  setMessages(newMessages);
  setIsStreaming(true);

  try {
    // Check if we have images - use vision API
    const hasImages = userMessage.images && userMessage.images.length > 0;

    if (hasImages) {
      // Use vision API
      const { claudeAPIService } = await import('@/services/claudeAPIService');

      const claudeMessages = newMessages.map(m => ({
        role: m.role,
        content: m.images && m.images.length > 0
          ? [
              ...m.images.map(img => ({
                type: 'image',
                source: {
                  type: 'base64',
                  media_type: 'image/png', // or detect from data URL
                  data: img.split(',')[1] // Remove data:image/png;base64, prefix
                }
              })),
              { type: 'text', text: m.content }
            ]
          : m.content
      }));

      const systemPrompt = isElement
        ? formatElementContextAsPrompt(initialContext)
        : formatPageContextAsPrompt(initialContext);

      const response = await claudeAPIService.sendMessage(
        claudeMessages,
        { system: systemPrompt, maxTokens: 4096 }
      );

      setMessages([...newMessages, { role: 'assistant', content: response }]);
    } else {
      // Use streaming text API (existing code)
      // ... existing streaming logic
    }
  } catch (error) {
    console.error('[InlineChatWindow] Error:', error);
    // ... error handling
  } finally {
    setIsStreaming(false);
  }
};

⚠️ Potential Issues & Mitigations ▶
Issue Mitigation
Large Image File Size Add validation (max 10MB) before adding to pending images. Show toast error for oversized files.
Base64 Encoding Overhead Base64 increases size by ~33%. Validate before conversion. Consider compression for large images.
API Key Not Configured Check apiConfigService.hasAPIKey() before API call. Show error message to configure key.
Vision API Costs Vision API costs more than text-only. Could add warning: "Image analysis costs more tokens".
Multiple Images in One Message Claude API supports multiple images. Map all pendingImages to content blocks.
Streaming with Vision Vision API may not support streaming. Use non-streaming sendMessage() when images present.
Media Type Detection Extract media type from data URL prefix (data:image/png;base64,) instead of hardcoding.
Message History Size Images stored as base64 in conversation history. Could exceed storage limits. Consider limiting history.

Message Format Evolution

Current (Text-Only)

{
  role: 'user',
  content: 'Hello'
}

New (With Images)

{
  role: 'user',
  content: 'What is this?',
  images: ['data:image/png;base64,...']

Claude API Content Block Transformation

When images are present, transform message to multimodal format:
// Input message
{
  role: 'user',
  content: 'Describe this image',
  images: ['data:image/png;base64,iVBORw0KG...']
}

// Transform to Claude API format
{
  role: 'user',
  content: [
    {
      type: 'image',
      source: {
        type: 'base64',
        media_type: 'image/png',
        data: 'iVBORw0KG...' // Without prefix
      }
    },
    { type: 'text', text: 'Describe this image' }
  ]
}

Media Type Detection Logic

function getMediaTypeFromDataUrl(dataUrl: string): string {
  const match = dataUrl.match(/data:([^;]+);/);
  return match ? match[1] : 'image/png';
}

function stripDataUrlPrefix(dataUrl: string): string {
  return dataUrl.split(',')[1];
}

Error Handling Strategy

  • Image Too Large: Show toast: "Image too large (max 10MB)"
  • Invalid Format: Show toast: "Unsupported image format"
  • API Error: Keep images in pending state, allow retry
  • Network Timeout: Show loading indicator, allow cancel
