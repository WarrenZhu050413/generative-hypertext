üìù Detailed Fixes

Fix 1: Container & Chat Window Styles

// Container (blocks page interaction, allows chat):
const container = document.createElement('div');
container.style.cssText = `
  position: fixed;
  top: 0;
  left: 0;
  width: 100%;
  height: 100%;
  z-index: 2147483646;
  pointer-events: none;  /* Blocks page clicks */
`;

// Chat window MUST have:
const chatWindow = ...;
chatWindow.style.pointerEvents = 'auto';  /* Accepts clicks */

Fix 2: Element Context Capture

interface ElementContext {
  // Element-specific data
  element: {
    tag: string;
    text: string;
    html: string;
    attributes: Record<string, string>;
    rect: DOMRect;
  };

  // Page context (optional)
  page?: {
    url: string;
    title: string;
  };
}

Fix 3: Context Menu Integration

// background/index.ts:
chrome.contextMenus.create({
  id: 'chat-with-element',
  title: 'üí¨ Chat about this with Claude',
  contexts: ['all']

chrome.contextMenus.onClicked.addListener((info, tab) => {
  if (info.menuItemId === 'chat-with-element') {
    chrome.tabs.sendMessage(tab.id, {
      type: 'OPEN_ELEMENT_CHAT',
      frameId: info.frameId  // Important for iframes!
    });
  }
});

Fix 4: Chat Window Positioning

// Position near element, not fixed location:
function positionChatNearElement(element) {
  const rect = element.getBoundingClientRect();

  // Position to right of element, or below if no space
  let x = rect.right + 20;
  let y = rect.top;

  // Keep on screen
  if (x + 420 > window.innerWidth) {
    x = rect.left - 420 - 20;
  }
  if (x < 0) {
    x = 20;
    y = rect.bottom + 20;
  }

  return { x, y };
}
