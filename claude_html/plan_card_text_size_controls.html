useFontSize Hook

// src/shared/hooks/useFontSize.ts
export function useFontSize() {
  const [fontSize, setFontSizeState] = useState<FontSize>('medium');
  const [fontSizeValues, setFontSizeValues] = useState<FontSizeValues>(
    getFontSizeValues('medium')
  );

  useEffect(() => {
    // Load from storage
    fontSizeService.getFontSize().then(size => {
      setFontSizeState(size);
      setFontSizeValues(getFontSizeValues(size));
    });

    // Listen for changes
    const handleChange = (changes: any, area: string) => {
      if (area === 'local' && changes.nabokov_font_size) {
        const newSize = changes.nabokov_font_size.newValue;
        setFontSizeState(newSize);
        setFontSizeValues(getFontSizeValues(newSize));
      }
    };
    chrome.storage.onChanged.addListener(handleChange);

    return () => chrome.storage.onChanged.removeListener(handleChange);
  }, []);

  const setFontSize = async (size: FontSize) => {
    await fontSizeService.setFontSize(size);
    setFontSizeState(size);
    setFontSizeValues(getFontSizeValues(size));
  };

  return { fontSize, fontSizeValues, setFontSize };
}
