Issues in v1

  • Z-index overwritten: bringToFront() clobbers inline z-index
  • Auto-pin missing: Parent not pinned when child opens

  • Coordinate z-index: Advance currentZIndex before setActive()
  • Auto-pin implemented: Pin parent in applyHypertext, track prior state

Step 2: Detect Parent from Selection Context

Add Helper Function (before applyHypertext)

function getParentSessionFromSelection(activeRange, sessions) {
  if (!activeRange) return null;

  // Walk up from selection to find parent tooltip
  let el = activeRange.commonAncestorContainer;
  if (el.nodeType === Node.TEXT_NODE) {
    el = el.parentElement;
  }

  const parentTooltip = el?.closest('.hx-chat-tooltip');
  if (!parentTooltip) return null;

  // Find session matching this tooltip
  const parentSessionId = parentTooltip.dataset.sessionId;
  return parentSessionId ? sessions.get(parentSessionId) : null;
}

function applyHypertext(instructions) {
  if (!activeRange || !activeSelection) return;

  // Detect parent session
  const parentSession = getParentSessionFromSelection(activeRange, sessions);

  // Check nesting depth
  const nestingLevel = parentSession ? parentSession.nestingLevel + 1 : 0;
  if (nestingLevel >= 5) {
    const shouldOpenChatPage = win.confirm(
      'Maximum nesting depth (5) reached. Open in chat page instead?'
    );
    if (shouldOpenChatPage) {
      openChatPageFromSession(parentSession || createTempSession(), null);
    }
    return;
  }

  // Auto-pin parent tooltip to prevent it from closing
  if (parentSession && !parentSession.isPinned) {
    session.parentWasPinned = false; // Was NOT pinned before
    const parentController = controllerFor(parentSession);
    if (parentController) {
      parentController.setPinned(true);
      console.log('[Hypertext] Auto-pinned parent tooltip');
    }
  } else if (parentSession) {
    session.parentWasPinned = true; // Was already pinned
  }

  // Calculate target z-index before creating tooltip
  let targetZIndex = TOOLTIP_BASE_Z_INDEX;
  if (parentSession) {
    const parentController = controllerFor(parentSession);
    if (parentController) {
      const parentZ = parseInt(
        parentController.element.style.zIndex || TOOLTIP_BASE_Z_INDEX,
        10
      );
      targetZIndex = parentZ + 1;
    }
  }

  // Advance global currentZIndex to ensure no conflicts
  if (targetZIndex > currentZIndex) {
    currentZIndex = targetZIndex;
  }

  // ... existing wrapper creation code

  const session = {
    // ... existing fields
    parentSessionId: parentSession?.id || null,
    childSessionIds: new Set(),
    nestingLevel,
    parentWasPinned: parentSession?.isPinned || false
  };

  // Link parent/child
  if (parentSession) {
    parentSession.childSessionIds.add(sessionId);
  }

  sessions.set(sessionId, session);

  // ... rest of existing code
}

Step 4: Lifecycle - Restore Parent Pin State on Close

Update closeChat Function (line ~2081)

function closeChat({ force = false, sessionId = null } = {}) {
  const targetSession = sessionId ? sessions.get(sessionId) : activeChatSession;
  if (!targetSession) return;

  if (!force && targetSession.isPinned) {
    return;
  }

  // NEW: Close all child sessions first
  if (targetSession.childSessionIds.size > 0) {
    targetSession.childSessionIds.forEach(childId => {
      closeChat({ force: true, sessionId: childId });
    });
    targetSession.childSessionIds.clear();
  }

  // NEW: Restore parent's pin state if this was a nested tooltip
  if (targetSession.parentSessionId) {
    const parentSession = sessions.get(targetSession.parentSessionId);
    if (parentSession) {
      // Remove this child from parent's tracking
      parentSession.childSessionIds.delete(targetSession.id);

      // Restore parent's original pin state
      if (!targetSession.parentWasPinned && parentSession.isPinned) {
        const parentController = controllerFor(parentSession);
        if (parentController) {
          parentController.setPinned(false);
          console.log('[Hypertext] Restored parent pin state (unpinned)');
        }
      }
    }
  }

  // ... existing cleanup code
  const controller = controllerFor(targetSession);
  if (controller) {
    controller.cancelClose();
    controller.hide();
    controller.setActive(false);
    if (force || !targetSession.isPinned) {
      controller.destroy();
      controllers.delete(targetSession.id);
    }
  }

  // ... rest unchanged
}

test('recursive hypertext with auto-pin', async ({ page }) => {
  await page.goto('demo/hypertext_navigation_demo.html');

  // Level 1: Generate root hypertext
  await selectText(page, 'first paragraph');
  await page.keyboard.press('Meta+Shift+K');
  await page.click('button:has-text("Generate")');
  await page.waitForSelector('.hx-chat-tooltip');

  const tooltip1 = page.locator('.hx-chat-tooltip').first();

  // Verify tooltip1 is not pinned initially
  let pinned1 = await tooltip1.getAttribute('data-pinned');
  expect(pinned1).toBe('false');

  // Level 2: Generate nested hypertext inside tooltip1
  await tooltip1.hover();
  await selectTextInTooltip(page, tooltip1, 'some word');
  await page.keyboard.press('Meta+Shift+K');
  await page.click('button:has-text("Generate")');
  await page.waitForSelector('.hx-chat-tooltip[data-nested="true"]');

  const tooltip2 = page.locator('.hx-chat-tooltip[data-nested="true"]').first();

  // Verify parent is now auto-pinned
  pinned1 = await tooltip1.getAttribute('data-pinned');
  expect(pinned1).toBe('true');

  // Verify z-index stacking
  const z1 = await tooltip1.evaluate(el =>
    parseInt(window.getComputedStyle(el).zIndex)
  );
  const z2 = await tooltip2.evaluate(el =>
    parseInt(window.getComputedStyle(el).zIndex)
  );
  expect(z2).toBe(z1 + 1);
  expect(z2).toBeGreaterThan(z1);

  // Close child, verify parent unpins
  await tooltip2.locator('[data-action="close"]').click();
  await page.waitForTimeout(200);

  pinned1 = await tooltip1.getAttribute('data-pinned');
  expect(pinned1).toBe('false');
});

‚ö†Ô∏è Edge Cases & Failure Modes ‚ñ∂

Case: Parent Already Pinned

Scenario: User manually pins parent, then generates nested hypertext

Solution: Store parentWasPinned = true, don't unpin on child close ‚úì

Case: Multiple Children

Scenario: Parent has 2+ child tooltips open

Solution: Only unpin when childSessionIds.size === 0 AND !parentWasPinned

Case: Force Close Parent

Scenario: User closes parent while children open

Solution: closeChat() cascade deletes all children first ‚úì

Case: Escape Key

Scenario: User presses Esc with nested tooltips

Solution: Existing handleEscapeKey closes activeChatSession (deepest) ‚úì

Case: Selection Spans Tooltips

Scenario: User drags selection from parent into child

Solution: commonAncestorContainer.closest() finds deepest tooltip ‚úì

Case: Z-Index Race

Scenario: Two nested tooltips created simultaneously

Solution: Advance currentZIndex synchronously before createController ‚úì
