const handleMouseMove = (e: MouseEvent) => {
  // Get element at cursor position
  const target = e.target as HTMLElement;

  // Ignore overlay elements
  if (target.closest('[data-nabokov-overlay]')) return;

  // Ignore non-visual elements
  const ignored = ['SCRIPT', 'STYLE', 'LINK', 'META', 'HEAD'];
  if (ignored.includes(target.tagName)) return;

  setHoveredElement(target);

  // Update element info for tooltip
  const rect = target.getBoundingClientRect();
  setElementInfo({
    tagName: target.tagName.toLowerCase(),
    classes: Array.from(target.classList),
    dimensions: {
      width: Math.round(rect.width),
      height: Math.round(rect.height)
    },
    position: { x: e.clientX, y: e.clientY }
  });
};

Single Click: Immediate Capture (Lines 316-478)

const handleClick = async (e: MouseEvent) => {
  e.preventDefault();
  e.stopPropagation();

  const target = e.target as HTMLElement;

  // Multi-select mode: Cmd/Ctrl+Click
  if (e.metaKey || e.ctrlKey) {
    toggleSelection(target);
    return;
  }

  // Single-click: immediate capture
  setIsCapturing(true);

  // 1. Sanitize HTML content
  const htmlContent = sanitizeHTML(target.outerHTML);

  // 2. Extract relevant styles
  const styles = extractRelevantStyles(target);

  // 3. Generate CSS selector
  const selector = generateSelector(target);

  // 4. Get surrounding context
  const context = target.parentElement
    ? sanitizeHTML(target.parentElement.outerHTML)
    : undefined;

  // 5. Create card object
  const card: Card = {
    id: generateId(),
    content: htmlContent,
    metadata: { /* url, title, dimensions, etc */ },
    stashed: stashImmediately,
    // ... other fields
  };

  // 6. Save to chrome.storage.local
  await saveCard(card);

  // 7. Broadcast update
  await chrome.runtime.sendMessage({
    type: 'CARD_STASHED',
    cardId: card.id
  });

  // 8. Auto-close after 2 seconds
  setTimeout(() => onClose(), 2000);
};

Why Shadow DOM?

â–¶Event Listener Setup & Cleanup

useEffect Hook (Lines 483-503)

useEffect(() => {
  if (!isActive) return;

  // Register listeners with capture phase (true)
  window.addEventListener('keydown', handleKeyDown, true);
  window.addEventListener('mousemove', handleMouseMove, true);
  window.addEventListener('click', handleClick, true);

  // Cleanup function
  return () => {
    window.removeEventListener('keydown', handleKeyDown, true);
    window.removeEventListener('mousemove', handleMouseMove, true);
    window.removeEventListener('click', handleClick, true);
  };
}, [isActive, handleKeyDown, handleMouseMove, handleClick]);

Why Capture Phase (true parameter)?

  • Events intercepted before reaching page elements
  • Allows preventDefault() to block page interactions during capture
  • Ensures overlay has first chance to handle events
