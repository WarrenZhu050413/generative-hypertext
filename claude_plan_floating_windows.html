üí¨ Floating Windows with State Persistence

Objective

Implement floating window system for all canvas cards with minimize/maximize functionality and full state persistence (input text, scroll position, conversation history).

Approach

Create reusable FloatingWindow component with WindowManager service. Every card can open as a window. Windows persist state across minimize/maximize/close/reopen.

Complexity: Medium
Est. Time: 1 week
Key Requirement: When windows are minimized (collapsed to title bar), ALL state must persist: chat input text, scroll position, conversation messages, any form data. Reopening the window restores exactly where the user left off.

‚ö†Ô∏è Potential Issues & Mitigations ‚ñ∂

Issue 1: State Capture Complexity

Capturing ALL input state (including dynamically added fields) is tricky

Mitigation:

  • Use FormData API to automatically capture all inputs within window
  • Store as Record<string, string> in windowState
  • On maximize, iterate and restore all values

Issue 2: Memory with Many Windows

Keeping many windows mounted (even if minimized) uses memory

Mitigation:

  • Limit max open windows to 20
  • Show warning at 15 windows
  • Offer "Close All Minimized" button
  • Consider lazy-loading window content (future optimization)

Issue 3: Window Overflow Off-Screen

Windows can be dragged off-screen and lost

Mitigation:

  • Constrain drag bounds to canvas viewport
  • Add "Reset Position" button in window header
  • On window open, auto-position with cascade offset

Issue 4: React State vs DOM State

Input values in DOM might not sync with React state

Mitigation:

  • Use controlled components for critical inputs (chat input)
  • Store in React state AND windowState
  • On minimize, read from DOM as fallback

Issue 5: Z-Index Wars with React Flow

React Flow uses z-index for nodes, may conflict

Mitigation:

  • Use z-index range 1000+ for windows (React Flow uses lower)
  • Floating windows container should be sibling to React Flow, not child

üîß WindowManager API ‚ñ∂ 
// src/services/windowManager.ts
import { WindowState } from '@/types/window';

class WindowManager {
  private windows = new Map<string, WindowState>();
  private maxZIndex = 1000;
  private listeners = new Set<() => void>();

  /**
   * Open a new window for the given card
   * Auto-positions with cascade offset
   */
  openWindow(cardId: string, card: Card): WindowState {
    const windowId = `window-${Date.now()}-${Math.random()}`;

    // Calculate cascade position
    const existingWindows = Array.from(this.windows.values());
    const offset = existingWindows.length * 30;

    const windowState: WindowState = {
      id: windowId,
      cardId,
      position: { x: 100 + offset, y: 100 + offset },
      size: { width: 500, height: 600 },
      isMinimized: false,
      zIndex: ++this.maxZIndex,
      chatInput: '',
      scrollPosition: 0,
      conversationMessages: card.conversation || [],
      formData: {},
      createdAt: Date.now(),
      lastInteractedAt: Date.now()
    };

    this.windows.set(windowId, windowState);
    this.notifyListeners();
    this.saveToStorage();

    return windowState;
  }

  /**
   * Close window and remove from state
   */
  closeWindow(windowId: string): void {
    this.windows.delete(windowId);
    this.notifyListeners();
    this.saveToStorage();
  }

  /**
   * Toggle minimize state
   */
  minimizeWindow(windowId: string): void {
    const window = this.windows.get(windowId);
    if (!window) return;

    window.isMinimized = true;
    window.lastInteractedAt = Date.now();
    this.notifyListeners();
    this.saveToStorage();
  }

  /**
   * Toggle maximize state
   */
  maximizeWindow(windowId: string): void {
    const window = this.windows.get(windowId);
    if (!window) return;

    window.isMinimized = false;
    window.lastInteractedAt = Date.now();
    this.notifyListeners();
  }

  /**
   * Update window state (position, input, etc.)
   */
  updateWindowState(windowId: string, updates: Partial<WindowState>): void {
    const window = this.windows.get(windowId);
    if (!window) return;

    Object.assign(window, updates);
    window.lastInteractedAt = Date.now();
    this.notifyListeners();

    // Debounced save
    this.debouncedSave();
  }

  /**
   * Bring window to front (highest z-index)
   */
  bringToFront(windowId: string): void {
    const window = this.windows.get(windowId);
    if (!window) return;

    window.zIndex = ++this.maxZIndex;
    this.notifyListeners();
  }

  /**
   * Get all windows
   */
  getAllWindows(): WindowState[] {
    return Array.from(this.windows.values())
      .sort((a, b) => a.zIndex - b.zIndex);
  }

  /**
   * Subscribe to window changes
   */
  subscribe(callback: () => void): () => void {
    this.listeners.add(callback);
    return () => this.listeners.delete(callback);
  }

  private notifyListeners(): void {
    this.listeners.forEach(listener => listener());
  }

  private debouncedSave = debounce(() => {
    this.saveToStorage();
  }, 500);

  async saveToStorage(): Promise<void> {
    const windowsArray = Array.from(this.windows.values());
    await chrome.storage.local.set({
      nabokov_windows: windowsArray
    });
  }

  async loadFromStorage(): Promise<void> {
    const result = await chrome.storage.local.get(['nabokov_windows']);
    const windowsArray = result.nabokov_windows || [];

    this.windows.clear();
    windowsArray.forEach((w: WindowState) => {
      this.windows.set(w.id, w);
      this.maxZIndex = Math.max(this.maxZIndex, w.zIndex);
    });

    this.notifyListeners();
  }
}

// Singleton instance
export const windowManager = new WindowManager();

// Utility
function debounce(fn: Function, ms: number) {
  let timer: number;
  return (...args: any[]) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), ms);
  };
}

const handleMinimize = () => {
  // Capture current state before minimizing
  const formData = captureFormData();
  const scrollPos = contentRef.current?.scrollTop || 0;

  windowManager.updateWindowState(windowState.id, {
    chatInput: chatInputRef.current?.value || '',
    scrollPosition: scrollPos,
    formData
  });

function captureFormData() {
  const form = contentRef.current;
  if (!form) return {};

  const formData = new FormData(form);
  const data: Record<string, string> = {};

  for (const [key, value] of formData.entries()) {
    data[key] = value.toString();
  }

  return data;
}
