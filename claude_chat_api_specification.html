File: src/services/chatService.ts

class ChatService {
  private conversations = new Map<string, Message[]>();
  private abortControllers = new Map<string, AbortController>();

  async *sendMessage(
    cardId: string,
    userMessage: string,
    cardContent: string
  ): AsyncGenerator<string> {
    // Add user message
    const userMsg: Message = {
      id: generateId(),
      role: 'user',
      content: userMessage,
      timestamp: Date.now()
    };
    this.addMessage(cardId, userMsg);

    // Create assistant message
    const assistantMsg: Message = {
      id: generateId(),
      role: 'assistant',
      content: '',
      timestamp: Date.now(),
      streaming: true,
      parentId: userMsg.id
    };
    this.addMessage(cardId, assistantMsg);

    // Stream response
    const controller = new AbortController();
    this.abortControllers.set(cardId, controller);

    try {
      const stream = await mockContentGenerator.generate(
        userMessage,
        cardContent,
        controller.signal
      );

      for await (const chunk of stream) {
        assistantMsg.content += chunk;
        yield chunk;
      }

      assistantMsg.streaming = false;
      this.saveConversation(cardId);
    } finally {
      this.abortControllers.delete(cardId);
    }
  }

  stopStreaming(cardId: string): void {
    this.abortControllers.get(cardId)?.abort();
  }

  getConversation(cardId: string): Message[] {
    return this.conversations.get(cardId) || [];
  }

  clearConversation(cardId: string): void {
    this.conversations.delete(cardId);
    this.saveConversation(cardId);
  }

  private addMessage(cardId: string, msg: Message): void {
    const conv = this.conversations.get(cardId) || [];
    conv.push(msg);
    this.conversations.set(cardId, conv);
  }

  private async saveConversation(cardId: string): Promise<void> {
    const card = await getCardById(cardId);
    card.conversation = this.getConversation(cardId);
    await saveCard(card);
  }
}

üíæ State Persistence ‚ñ∂

Auto-Save Strategy

  • Chat Input: Debounced save every 500ms to windowState.chatInput
  • Messages: Save to card.conversation immediately after each message completes
  • Scroll Position: Save on scroll end (debounced 300ms)
  • Minimize/Maximize: Preserve all state, no unmounting

Load on Window Open

const loadWindowState = (cardId: string): WindowState => {
  const card = getCardById(cardId);
  const savedState = windowStorage.get(cardId);

  return {
    cardId,
    position: savedState?.position || { x: 100, y: 100 },
    size: savedState?.size || { width: 600, height: 500 },
    minimized: savedState?.minimized || false,
    zIndex: getNextZIndex(),

    // Restore chat state
    chatInput: savedState?.chatInput || '',
    conversationMessages: card.conversation || [],
    scrollPosition: savedState?.scrollPosition || 0,
    isStreaming: false
  };
};

Save on State Change

stopStreaming(cardId) Abort ongoing stream for card
getConversation(cardId) Get all messages for card
clearConversation(cardId) Delete all messages for card
FloatingWindowChat messages: Message[] Conversation history to display
currentInput: string Current input field value
onSendMessage(text) Callback when user sends message
onInputChange(value) Callback when input changes
onStopStreaming() Callback to stop streaming response

conversationMessages: Message[] Full conversation loaded from card
isStreaming: boolean Whether Claude is currently responding
