Standard API Selection Pattern

// Import both services
import { apiConfigService } from './apiConfig';
import { claudeAPIService } from './claudeAPIService';
import { mockContentGenerator } from './mockContentGenerator';

// In async function
async function generateContent(...) {
  const hasAPIKey = await apiConfigService.hasAPIKey();

  if (hasAPIKey) {
    // Use real Claude API
    const response = await claudeAPIService.sendMessage([
      { role: 'user', content: prompt }
    ], {
      system: systemPrompt,
      maxTokens: 4096
    });
    return response;
  } else {
    // Fallback to mock (development)
    console.warn('[Service] No API key, using mock');
    return await mockContentGenerator.generate(...);
  }
}
